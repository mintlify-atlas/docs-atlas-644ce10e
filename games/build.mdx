Build is a creative sandbox game where players can freely build structures using voxel blocks, fly around, and interact with NPCs like pigs and sharks.

## Gameplay

- **Mode**: Creative building sandbox
- **Blocks**: Marble voxels with color customization
- **Flight**: Toggle flying with F key
- **Water**: Swimming mechanics
- **NPCs**: Pigs on land, sharks in water
- **Tools**: Blaster and dagger weapons
- **Multiplayer**: Collaborative building

## Systems

### BuildSystem

Location: `core/src/games/build/Build.ts:70`

```typescript
const BuildSystem = SystemBuilder({
  id: "BuildSystem",
  init: (world) => {
    // Load pre-built map
    world.blocks.loadMap(BuildMap)
    world.blocks.coloring = BuildMapColoring

    const mobileUI = MobileUI(world)

    // Set initial camera rotation
    if (world.client) {
      world.client.controls.localAim.x = PI / 2 * 2.5
    }

    return {
      id: "BuildSystem",
      query: [],
      priority: 3,
      onTick: () => {
        const state = world.game.state as BuildState

        // Menu control
        if (world.client && !world.client.mobile) {
          world.client.menu = document.pointerLockElement === null
        }

        mobileUI?.update()

        const players = world.players()

        for (const player of players) {
          const character = player.components.controlling.getCharacter(world)
          if (!character) continue

          const { position } = character.components
          const { standing, z, flying } = position.data

          // Handle swimming
          if (flying) {
            position.data.swimming = false
          } else if (z < -0.2) {
            position.data.swimming = true
            // Can't double-jump while swimming
            if (!state.doubleJumped.includes(character.id)) {
              state.doubleJumped.push(character.id)
            }
          } else if (z >= 0) {
            position.data.swimming = false
          }

          // Reset double jump on landing
          if (standing) {
            state.doubleJumped = state.doubleJumped.filter(id => id !== character.id)
          }
        }
      }
    }
  }
})
```

```typescript
      }),
      networked: Networked(),
      inventory: Inventory([BlasterItem, DaggerItem]),
      collider: Collider({ shape: "ball", radius: 0.1 }),
      health: Health(),
      input: Input({
        joystick: ({ client }) => {
          // Mobile joystick with camera-relative movement
        },
        press: {
          "1": () => ({ actionId: "setActiveItemIndex", params: { index: 0 } }),
          "2": () => ({ actionId: "setActiveItemIndex", params: { index: 1 } }),
          "mb2": ({ hold, world, character }) => {
            // Place blocks
            const dir = world.three!.camera.dir(world)
            const camera = world.three!.camera.pos()
            const pos = character.components.position.xyz()
            const { blockColor } = world.settings<BuildSettings>()
            return { 
              actionId: "place", 
              params: { dir, camera, pos, type: 12, blockColor } 
            }
          },
          "scrolldown": ({ client, world }) => {
            // Cycle block color down
            const { blockColor } = world.settings<BuildSettings>()
            world.game.settings.blockColor = nextColor(blockColor)
          },
          "scrollup": ({ client, world }) => {
            // Cycle block color up
            const { blockColor } = world.settings<BuildSettings>()
            world.game.settings.blockColor = nextColor(blockColor, true)
          },
          "f": () => {
            // Toggle flying
            bob.components.position.data.flying = !bob.components.position.data.flying
          },
          "shift": () => {
            // Fly/swim down
            return { actionId: "down" }
          },
          " ": ({ hold }) => {
            // Jump or fly up
            const { flying, swimming } = bob.components.position.data
            if (flying || swimming) {
              return { actionId: "up" }
            } else {
              return { actionId: "jump", params: { hold } }
            }
          },
          "q": ({ world }) => {
            // Toggle first/third person
            const { camera } = world.three!
            camera.mode = camera.mode === "first" ? "third" : "first"
          },
          "n": ({ world }) => {
            // Toggle nametags
            const settings = world.settings<BuildSettings>()
            settings.showNametags = !settings.showNametags
          },
          "x": ({ character, world }) => {
            // Volume placement mode
            // Place multiple blocks at once
          },
          // WASD movement
          "w,a": ({ world }) => ({ actionId: "move", params: { key: "wa", ...upAndDir(world) } }),
          "w,d": ({ world }) => ({ actionId: "move", params: { key: "wd", ...upAndDir(world) } }),
          // ... other diagonal movements
        }
      }),
      actions: Actions({
        place: Place,
        point: Point,
        ready: Ready,
        setActiveItemIndex,
        up: Action("up", () => {
          // Fly/swim upward
          const { position } = bob.components
          let factor = 0.015
          if (position.data.swimming) {
            factor = 0.01 * (0.3 + position.submerged())
          }
          position.impulse({ z: factor })
        }),
        down: Action("down", () => {
          // Fly/swim downward
          const { position } = bob.components
          let factor = 0.015
          if (position.data.swimming) {
            factor = 0.01 * (1.3 - position.submerged())
          }
          position.impulse({ z: -factor })
        }),
        jump: Action("jump", ({ world, params }) => {
          // Jump with double-jump
          const { position } = bob.components
          if (position.data.flying) return
          if (bob.components.health?.dead()) return

          const state = world.state<BuildState>()

          if (!position.data.standing && state.doubleJumped.includes(bob.id)) return
          if (!position.data.standing && params.hold) return

          if (!position.data.standing) {
            position.setVelocity({ z: max(0.05, 0.025 + position.data.velocity.z) })
            state.doubleJumped.push(bob.id)
          } else {
            position.setVelocity({ z: 0.05 })
          }

          world.client?.sound.play({ name: "bubble", threshold: { pos: position.data, distance: 5 } })
        }),
        moveAnalog: Action("moveAnalog", ({ entity, params }) => {
          // Mobile joystick movement
        }),
        move: Action("move", ({ entity, params }) => {
```

### Volume Placement

Location: `core/src/games/build/Bob.ts:178`

Place multiple blocks at once:

```typescript
"x": ({ hold, world, character, client }) => {
  if (hold || !character || !wipMesh) return

  // Define cuboid from two corner points
  if (wipStart && wipEnd) {
    // Place all blocks in volume
    for (let x = wipStart.x; x !== wipEnd.x + xDir; x += xDir) {
      for (let y = wipStart.y; y !== wipEnd.y + yDir; y += yDir) {
        for (let z = wipStart.z; z !== wipEnd.z + zDir; z += zDir) {
          world.blocks.add({ x, y, z, type: 12 })
          world.blocks.coloring[`${x},${y},${z}`] = blockColor
        }
      }
    }
    client.sound.play({ name: "click2" })
    wipStart = undefined
  } else {
    wipStart = wipEnd  // Set first corner
  }
}
```
